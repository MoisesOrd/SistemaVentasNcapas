Tabla Cliente
SELECT CLIENTE.IDCLIENTE , PERSONA.NOMBRE, PERSONA.APELLIDO, CLIENTE.CODIGOCLIENTE, CLIENTE.ESTADO, PERSONA.CI
FROM     CLIENTE INNER JOIN
                  PERSONA ON CLIENTE.IDPERSONA = PERSONA.IDPERSONA
Tabla Usuario
SELECT USUARIO.IDUSUARIO, USUARIO.NOMBREUSER, PERSONA.NOMBRE, PERSONA.APELLIDO, PERSONA.CI, PERSONA.ESTADO
FROM     USUARIO INNER JOIN
                  PERSONA ON USUARIO.IDPERSONA = PERSONA.IDPERSONA
Tabla Venta
SELECT CLIENTE.IDCLIENTE, VENTA.FECHA, VENTA.TOTAL, CLIENTE.CODIGOCLIENTE
FROM     VENTA INNER JOIN
                  CLIENTE ON VENTA.IDCLIENTE = CLIENTE.IDCLIENTE
Tabla UsuarioRol
SELECT USUARIOROL.IDROL, ROL.NOMBRE AS PUESTO, USUARIO.NOMBREUSER, USUARIO.CONTRASEÃ‘A, USUARIO.FECHAREG, ROL.ESTADO

                  ROL ON USUARIOROL.IDROL = ROL.IDROL INNER JOIN

Tabla DetalleVenta
SELECT DETALLEVENTA.IDDETALLEVENTA, DETALLEVENTA.CANTIDAD, DETALLEVENTA.PRECIOVENTA, DETALLEVENTA.SUBTOTAL, VENTA.IDCLIENTE, VENTA.FECHA, DETALLEVENTA.ESTADO, PRODUCTO.NOMBRE, 
                  PRODUCTO.DESCRIPCION
FROM     DETALLEVENTA INNER JOIN
                  VENTA ON DETALLEVENTA.IDVENTA = VENTA.IDVENTA INNER JOIN
                  PRODUCTO ON DETALLEVENTA.IDPRODUCTO = PRODUCTO.IDPRODUCTO
Tabla Detalle Ingreso
SELECT DETALLEING.IDDETALLEING, DETALLEING.FECHAVENC, DETALLEING.CANTIDAD, DETALLEING.PRECIOCOSTO, DETALLEING.PRECIOVENTA, DETALLEING.SUBTOTAL, PRODUCTO.DESCRIPCION, DETALLEING.ESTADO, 
                  PRODUCTO.ESTADO AS ESTADO
FROM     DETALLEING INNER JOIN
                  PRODUCTO ON DETALLEING.IDPRODUCTO = PRODUCTO.IDPRODUCTO
Tabla Ingreso
SELECT INGRESO.IDINGRESO, INGRESO.FECHAINGRESO, INGRESO.TOTAL, PROVEEDOR.NOMBRE, INGRESO.ESTADO
FROM     INGRESO INNER JOIN
                  PROVEEDOR ON INGRESO.IDPROVEEDOR = PROVEEDOR.IDPROVEEDOR
Tabla Provee
SELECT PROVEE.IDPROVEE, PROVEE.FECHA, PROVEE.PRECIO, PRODUCTO.NOMBRE, PRODUCTO.UNIDAD, PRODUCTO.ESTADO, PROVEEDOR.IDPROVEEDOR, PROVEEDOR.NOMBRE AS Expr1
FROM     PROVEE INNER JOIN
                  PRODUCTO ON PROVEE.IDPRODUCTO = PRODUCTO.IDPRODUCTO INNER JOIN
                  PROVEEDOR ON PROVEE.IDPROVEEDOR = PROVEEDOR.IDPROVEEDOR
Tabla Producto
SELECT PRODUCTO.IDPRODUCTO, PRODUCTO.NOMBRE, PRODUCTO.UNIDAD, PRODUCTO.DESCRIPCION, MARCA.NOMBRE AS NombredeMarca, TIPOPROD.NOMBRE AS TipodeProdcuto
FROM     PRODUCTO INNER JOIN
                  MARCA ON PRODUCTO.IDMARCA = MARCA.IDMARCA INNER JOIN
                  TIPOPROD ON PRODUCTO.IDTIPOPROD = TIPOPROD.IDTIPOPROD
